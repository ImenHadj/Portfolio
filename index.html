Jenkinsfile

pipeline {
    agent any
    environment {
        DOCKERHUB = credentials('dockerHub') 
        IMAGE_TAG = "1.1.${BUILD_NUMBER}"
        SONAR = tool 'sonar-scanner'
       
    }

    stages {
        stage('GIT') {
            steps {
                git branch: 'Bloc', url: ''
            }
        }
        
        stage('MAVEN CLEAN') {
            steps {
                sh "mvn clean"
            }
        }
        
        stage('MAVEN COMPILE') {
            steps {
                sh "mvn compile"
            }
        }
        
        stage('JUNIT/MOCKITO') {
            steps {
                echo 'Test du code : ';
                sh "mvn test"
            }
        }
        
        
        stage('MVN SONARQUBE') {
            steps {
                echo 'Analyse de la qualité du code : ';
                withCredentials([ 
                    string(credentialsId: 'SonarURL', variable: 'Sonar_URL'), 
                    string(credentialsId: 'SonarID', variable: 'Sonar_ID') 
                ]) { 
                    
                    sh "mvn org.jacoco:jacoco-maven-plugin:0.8.11:report"
                    sh '''$SONAR/bin/sonar-scanner -X \
                        -Dsonar.host.url=${Sonar_URL} \
                        -Dsonar.token=${Sonar_ID} \
                        -Dsonar.projectName=tpfoyerDevops \
                        -Dsonar.java.binaries=target/classes \
                        -Dsonar.projectKey=tpfoyerDevops \
                        -Dsonar.jacoco.reportPaths=target/jacoco.exec \
                        -Dsonar.jacoco.reportPath=target/site/jacoco/jacoco.xml'''
                }

            }
        }
        
        stage('MVN PACKAGE') {
            steps {
                echo 'Package du code : ';
                sh "mvn package"
            }
        }
        
        stage('NEXUS') {
            steps {
                echo 'Deployment sur Nexus : ';
                sh "mvn deploy"
            }
        }
        
        stage('DOCKER IMAGE') {
            steps {
                echo 'Building docker image : ';
                sh "docker build  -t imenhadjyahya-g7-coconsult:1.1.${env.BUILD_NUMBER} ."
            }
        }
        
        stage("DOCKER HUB") {
            steps {
                echo "Connecting and Pushing Docker image to DockerHub : "
                sh 'echo $DOCKERHUB_PSW | docker login -u $DOCKERHUB_USR --password-stdin'
                sh "docker push imenhadjyahya-g7-coconsult:1.1.${env.BUILD_NUMBER}"
            }
        }
        
        stage('DOCKER COMPOSE') {
            steps {
                echo 'Starting Docker compose : ';
                sh 'docker compose up -d'
            }
        }
        
        stage('TRIVY SCAN') {
           steps {
               echo 'Scanning Docker image with Trivy : '
               sh 'trivy image --timeout 30m --format table --security-checks vuln --ignore-unfixed -o trivy-scan-report.html imenhadjyahya-g7-coconsult:1.1.${BUILD_NUMBER}'
               archiveArtifacts artifacts: 'trivy-scan-report.html', allowEmptyArchive: true
           }
}



 
    }
}
